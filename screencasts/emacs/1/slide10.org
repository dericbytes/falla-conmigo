 Usa %c para 0 y %C por 1 en los variables de formato.

 #+begin_src emacs-lisp :results silent
      (column-number-mode)
       
      (setq mode-line-position-column-line-format
 	                        '(" ZERO COL: %c LIN: %l"))

      (setq mode-line-position-column-line-format
 	                        '(" ONE COL: %C LIN: %l"))
 #+end_src
